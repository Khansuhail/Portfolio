 

   
// We declare a function named printLoopIndicesAndValues, it takes a parameter: array
function printLoopIndicesAndValues(array){
  // In the body of the function we create a for loop, we iterate through the loop with the local variable i.  The loop continues, incrementing the iterator, i until i is greater than or equal to the arrays length.
  for (var i = 0; i < array.length; i++) {
    // in each iteration of the loop, we console log, i, the index of the array, and array at i, the value of the array at that index.
    console.log(i, array[i]);
  }
  // we return 'undefined'
}
//
printLoopIndicesAndValues([1,4,9,24]);
